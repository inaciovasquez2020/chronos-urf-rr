\begin{proof}
Let $S\subseteq [n]$ be the support of $f$ with $|S|\le s$.
For any fixed assignment $a\in{0,1}^S$, the number of completions $b\in{0,1}^{[n]\setminus S}$
yielding total parity $0$ differs from the number yielding total parity $1$ by at most $1$,
and both are equal to $2^{n-|S|-1}$ unless $n-|S|=0$.
Thus the conditional distribution on $x_S$ under $\mu_0$ and under $\mu_1$ differs pointwise by at most
$2^{-(n-|S|)}$, hence the claimed bound.
The TV claim is immediate since $\mu_0,\mu_1$ have disjoint supports.
\end{proof}
\section*{Corrected Conditional Target}
\begin{definition}[Rigidity hypothesis schema]
A rigidity hypothesis is any explicit assumption on an instance family guaranteeing that global distinguishability cannot be confined to a single global parity-type constraint while all bounded-support tests remain nearly unbiased.
\end{definition}
\begin{conjecture}[Rigidity-conditional bounded-support witness extraction]
Fix $k,\Delta$. Assume an explicit rigidity hypothesis $\mathsf{Rig}(k,\Delta)$ for a bounded-degree instance family.
Then there exist constants $R,B,C$ depending only on $k,\Delta$ such that whenever two refinement histories are FO$^k$-distinguishable at some time $t$,
there exists a witness supported on a union of at most $B$ radius-$R$ patches with bias at least $2^{-C}$.
\end{conjecture}
