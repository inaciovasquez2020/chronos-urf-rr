\title{Cycle--Local Rigidity: Main Theorem}
\date{}
\maketitle

\section*{Definitions}

Let $G=(V,E)$ be a finite graph with maximum degree $\Delta$.

Define the local cycle-overlap rank

\[
C_R(G)=
\dim_{\mathbb F_2}
\left(
\sum_{v\in V}\iota_{v*}Z_1(B_R(v))
\right)
\]

and the number of FO$^k$ rooted $r$-types

\[
T_r(G)=
\left|
\{
\operatorname{tp}^k_r(v):v\in V
\}
\right|.
\]

\section*{Main Theorem}

\begin{theorem}[Cycle--Local Rigidity]
For fixed parameters $(k,\Delta,R,r)$ there exists a constant $c$
such that if

\[
C_R(G)\ge \alpha |V|
\]

for some $\alpha>0$, then

\[
T_r(G)\ge 2.
\]

\end{theorem}

\section*{Proof}

Assume for contradiction that

\[
T_r(G)=1.
\]

Then every vertex of $G$ has the same FO$^k$ $r$-type.

By FO$^k$ cycle-space invariance, the rooted $R$-balls
$B_R(v)$ are all isomorphic as graphs.

Hence their cycle spaces $Z_1(B_R(v))$ are identical.

Thus every vector contributing to

\[
\sum_{v\in V}\iota_{v*}Z_1(B_R(v))
\]

comes from a fixed bounded-dimensional family determined
by this single rooted ball type.

Since $B_R(v)$ contains at most $O(\Delta^R)$ edges, the
dimension of its cycle space is bounded by a constant
$c(k,\Delta,R)$.

Therefore

\[
C_R(G)\le c(k,\Delta,R).
\]

This contradicts the assumption

\[
C_R(G)\ge \alpha |V|.
\]

Hence

\[
T_r(G)\ge2.
\]

\end{proof}
