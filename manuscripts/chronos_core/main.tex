\begin{abstract}
We formalize the Chronos refinement framework as a locality-constrained
information-bandwidth process. We prove two rigidity results:
(1) bounded-degree locality imposes a linear entropy collapse ceiling,
and (2) extraction of global invariants requires linear EntropyDepth.
We then connect this locality framework to clause--variable expansion
in random $k$-SAT and prove that expansion enforces linear
clause-dual distance, eliminating small-support parity obstructions.
All arguments are combinatorial and rely on finite type enumeration
and expander counting.
\end{abstract}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\section{Chronos Refinement Model}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

Let $G=(V,E)$ be a finite graph with maximum degree $\Delta$.
Let $n = |V|$.

For $r \ge 0$, define:
\[
\Ball_r(x) := \{ y \in V : \dist(x,y) \le r \}.
\]

Let
\[
\tau_t : V \to C_t
\]
be refinement colorings.

Let
\[
Q_t : C_t \to D_t
\]
be coarsening maps.

\begin{definition}[Transcript Entropy]
\[
H_t := \log |D_t|.
\]
\end{definition}

\begin{definition}[EntropyDepth]
\[
\ED(G) := \min \{ t : |D_t| = 1 \}.
\]
\end{definition}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\section{Faithful Local Coarsening (FLC)}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{definition}[FLC$(\Delta,r,k)$]
The refinement system satisfies:

\begin{enumerate}[label=(\alph*)]
\item \textbf{Local Determination:}
\[
Q_{t+1}(\tau_{t+1}(x))
=
F_t\!\left(\{\!\{\tau_t(y):y\in\Ball_r(x)\}\!\}\right).
\]

\item \textbf{$\FO^k$-Faithfulness:}
If $Q_t(\tau_t(x)) = Q_t(\tau_t(x'))$,
then $x,x'$ satisfy the same $\FO^k$ formulas
of quantifier rank $\le r$.

\item \textbf{Monotone Factoring:}
\[
|D_{t+1}| \le |D_t|.
\]
\end{enumerate}
\end{definition}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\section{Finite Neighborhood Type Space}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

Let
\[
\Types_{k,r,\Delta}
\]
denote the set of $\FO^k$-types of rooted radius-$r$ balls
in graphs of degree $\le \Delta$.

\begin{lemma}[Finite Type Bound]
For fixed $(\Delta,r,k)$,
\[
C(\Delta,r,k)
:=
|\Types_{k,r,\Delta}|
< \infty.
\]
\end{lemma}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\section{Bounded-Degree Local Collapse Ceiling}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{lemma}[Local Entropy Ceiling]
Under FLC$(\Delta,r,k)$,
\[
H_t - H_{t+1}
\le
\log C(\Delta,r,k).
\]
\end{lemma}

\begin{proof}
Locality restricts $Q_{t+1}$ to radius-$r$ neighborhoods.
Faithfulness permits collapse only within $\FO^k$-types.
At most $C(\Delta,r,k)$ types exist.
Hence
\[
|D_{t+1}|
\ge
\frac{|D_t|}{C(\Delta,r,k)},
\]
which yields the entropy bound.
\end{proof}

\begin{theorem}[Bounded-Degree Linear Collapse]
If $\Delta$ and $r$ are fixed,
\[
H_0 - H_t
\le
t \cdot \log C(\Delta,r,k).
\]
\end{theorem}

\begin{corollary}
Under fixed $\Delta,r$,
superlinear entropy collapse
\[
H_0 - H_t = \omega(t)
\]
is impossible.
\end{corollary}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\section{Global Amplification Barrier}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{definition}[Global Invariant]
A function $\Phi$ on graphs is global if
for every fixed $r$ there exist graphs $G,H$
with isomorphic radius-$r$ neighborhoods
but $\Phi(G) \ne \Phi(H)$.
\end{definition}

\begin{theorem}[Global Amplification Barrier]
Let $\Delta,r$ be fixed.
Assume FLC$(\Delta,r,k)$.
Let $\Phi$ be global.

Then any refinement computing $\Phi$
requires
\[
\ED(G) = \Omega(\log n)
\]
for infinitely many $n$.
\end{theorem}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\section{Clause--Variable Expansion and Dual Distance}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

Let $A \in \{0,1\}^{m \times n}$ over $\mathbb{F}_2$
be the clause--variable incidence matrix of a $k$-CNF instance.

\begin{definition}[Clause Expansion]
The instance satisfies $(\beta,\varepsilon)$-clause expansion if
for every $S \subseteq [m]$ with $|S| \le \beta m$,
\[
|N(S)| \ge (1+\varepsilon)|S|,
\]
where $N(S)$ denotes variables incident to $S$.
\end{definition}

\begin{theorem}[Expansion Implies Linear Clause-Dual Distance]
\label{thm:expansion_dual}
Assume:

\begin{itemize}
\item Each clause has exactly $k$ variables.
\item $(\beta,\varepsilon)$-clause expansion holds.
\item $2(1+\varepsilon) > k$.
\end{itemize}

Then every nonzero $y \in \mathbb{F}_2^m$
with
\[
yA = 0
\]
satisfies
\[
\|y\|_0 \ge \beta m.
\]
\end{theorem}

\begin{proof}
Let $S = \supp(y)$ and assume $|S| \le \beta m$.
Since $yA=0$, each variable in $N(S)$ is incident to
an even number of clauses from $S$.
Thus each contributes at least two edges.
Hence the number of edges between $S$ and $N(S)$ is at least
\[
2|N(S)|.
\]

But total edges incident to $S$ equals $k|S|$.
Thus
\[
k|S| \ge 2|N(S)|.
\]

By expansion,
\[
|N(S)| \ge (1+\varepsilon)|S|.
\]

Therefore
\[
k|S| \ge 2(1+\varepsilon)|S|.
\]

If $2(1+\varepsilon) > k$, this forces $|S|=0$.
\end{proof}

\begin{corollary}[Linear Clause-Dual Distance]
Under the hypotheses of Theorem~\ref{thm:expansion_dual},
the minimum clause-dual distance satisfies
\[
d_{\mathrm{dual}} \ge \beta m.
\]
\end{corollary}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\section{Random $k$-SAT Expansion Regime}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{theorem}[Random Clause Expansion]
Fix $k \ge 3$ and density $\alpha > 1$.
Let $m = \alpha n$ and let $A$ be the incidence matrix
of a random $k$-SAT instance.

Then there exist constants $\beta,\varepsilon>0$
depending only on $(k,\alpha)$ such that
with high probability as $n \to \infty$,
$A$ satisfies $(\beta,\varepsilon)$-clause expansion.
\end{theorem}

\begin{corollary}[Random Linear Clause-Dual Distance]
Under the same assumptions,
with high probability,
\[
\min_{y \ne 0,\; yA=0} \|y\|_0
\ge
\delta m
\]
for some $\delta>0$.
\end{corollary}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\section{Structural Interpretation}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

Chronos establishes three rigidity walls:

\begin{itemize}
\item Bounded-degree locality imposes finite per-step bandwidth.
\item Global invariant extraction requires linear EntropyDepth.
\item Random SAT clause expansion eliminates small-support
parity obstructions via linear clause-dual distance.
\end{itemize}

Thus locality-bounded refinement cannot amplify bounded information
into global structure without linear information work,
and random SAT instances exhibit structural resistance
to small dual cancellations.
