\section{Conflict–Cycle Production Theorem}

\begin{theorem}
Let $F$ be a bounded-degree CNF formula with incidence graph $G_F$.

If
\[
CR(G_F)=\Omega(n)
\]

then every resolution refutation of $F$ contains a timestep $t$ where

\[
r_{\mathrm{sim}}(t)=\Omega(n)
\]

where $r_{\mathrm{sim}}(t)$ is the GF(2) span of the cuts induced by the
live clauses at time $t$.
\end{theorem}
